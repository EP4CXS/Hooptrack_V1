${d.team} Stats (Starters)

Player 2PT 3PT FT REB AST STL BLK TO FOUL PTS
${row.playerName} ${row.fouledOut ? 'FOUL OUT' : ""} ${row.fgm2}/${row.fga2} ${row.fgm3}/${row.fga3} ${row.ftm}/${row.fta} ${Number(row.oreb || 0) + Number(row.dreb || 0)} ${row.assists || 0} ${row.steals || 0} ${row.blocks || 0} ${row.turnovers || 0} ${row.fouls || 0} ${row.points || 0}

`; }; const ensureSelectedIsActive = (state) => { const d = sideData(state, selectedPlayer.side); if (!d.active.includes(selectedPlayer.rosterIndex) || !d.roster[selectedPlayer.rosterIndex]) { const home = activeRows(state, "home")[0]; const away = activeRows(state, "away")[0]; const next = home || away; selectedPlayer = next ? { side: home ? No pending or in-progress games for selected bracket.

'; return; } const matchup = matchups.find((m) => String(m.id) === selectedMatchupId) || matchups[0]; selectedMatchupId = String(matchup.id); if (!matchup) return; const state = await ensureGameState(bracket, matchup); ensureSelectedIsActive(state); const panel = document.getElementById("game-tracker-panel"); const meta = document.getElementById("games-header-meta"); if (meta) { meta.textContent = `${bracket.name} • ${matchups.length} active matchup${matchups.length > 1 ? "s" : ""}`; } const homeLogo = logoByTeam[matchup.team1]; const awayLogo = logoByTeam[matchup.team2]; const selectedRow = sideData(state, selectedPlayer.side).roster[selectedPlayer.rosterIndex]; panel.innerHTML = `
${homeLogo ? `` : `${String(matchup.team1).slice(0, 2).toUpperCase()}`}
${matchup.team1}
${state.score1}
VS
Q${state.quarter} • ${state.clock}
${matchup.team2}
${state.score2}
${awayLogo ? `` : `${String(matchup.team2).slice(0, 2).toUpperCase()}`}

Selected: ${selectedRow ? `${selectedRow.playerName} (${selectedPlayer.side === "home" ? matchup.team1 : matchup.team2})` : "None"}

${[ ["twoMade", "2PT Made"], ["twoMiss", "2PT Miss"], ["threeMade", "3PT Made"], ["threeMiss", "3PT Miss"], ["ftMade", "FT Made"], ["ftMiss", "FT Miss"], ["oreb", "OREB"], ["dreb", "DREB"], ["assist", "AST"], ["steal", "STL"], ["block", "BLK"], ["turnover", "TO"], ["foul", "Foul"], ["sub", "Substitution"], ["foulOut", "Foul Out"], ].map(([id, label]) => ``).join("")}
${renderPlayerTable(state, "home")} ${renderPlayerTable(state, "away")}
This will finalize matchup result and persist player stats.
`; panel.querySelectorAll("[data-player-row]").forEach((rowEl) => { rowEl.addEventListener("click", () => { const [side, idx] = rowEl.getAttribute("data-player-row").split("|"); selectedPlayer = { side, rosterIndex: Number(idx) }; renderTracker(); }); }); const saveSnapshot = () => { actionHistory.push({ matchupId: matchup.id, snapshot: JSON.parse(JSON.stringify({ ...state, selectedPlayer })), }); }; panel.querySelectorAll("[data-action]").forEach((btn) => { btn.addEventListener("click", () => { const action = btn.getAttribute("data-action"); if (action === "undo") { const last = actionHistory.pop(); if (!last || String(last.matchupId) !== String(matchup.id)) return; Object.assign(state, last.snapshot); selectedPlayer = last.snapshot.selectedPlayer; renderTracker(); return; } if (action === "prevQuarter" || action === "nextQuarter") { saveSnapshot(); state.quarter = action === "prevQuarter" ? Math.max(1, Number(state.quarter || 1) - 1) : Math.min(4, Number(state.quarter || 1) + 1); renderTracker(); return; } const d = sideData(state, selectedPlayer.side); const row = d.roster[selectedPlayer.rosterIndex]; if (!row) return; saveSnapshot(); if (action === "twoMade") { applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fgm2"); applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fga2"); } if (action === "twoMiss") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fga2"); if (action === "threeMade") { applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fgm3"); applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fga3"); } if (action === "threeMiss") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fga3"); if (action === "ftMade") { applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "ftm"); applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fta"); } if (action === "ftMiss") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fta"); if (action === "oreb") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "oreb"); if (action === "dreb") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "dreb"); if (action === "assist") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "assists"); if (action === "steal") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "steals"); if (action === "block") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "blocks"); if (action === "turnover") applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "turnovers"); if (action === "foul") { applyStat(state, selectedPlayer.side, selectedPlayer.rosterIndex, "fouls"); if (Number(row.fouls || 0) >= 6) { row.fouledOut = true; pickSubPlayer(state, selectedPlayer.side, selectedPlayer.rosterIndex).then(() => renderTracker()); } } if (action === "sub") { pickSubPlayer(state, selectedPlayer.side, selectedPlayer.rosterIndex).then(() => renderTracker()); } if (action === "foulOut") { row.fouls = Math.max(6, Number(row.fouls || 0)); row.fouledOut = true; pickSubPlayer(state, selectedPlayer.side, selectedPlayer.rosterIndex).then(() => renderTracker()); } renderTracker(); }); }); panel.querySelector("#game-save-form").addEventListener("submit", async (e) => { e.preventDefault(); const fresh = await fetchJson(`${apiBase}/brackets/${bracket.id}/`); const updatedMatchups = fresh.matchups.map((m) => ( String(m.id) === String(matchup.id) ? { ...m, score1: state.score1, score2: state.score2, winner: state.score1 >= state.score2 ? m.team1 : m.team2, status: "completed" } : m )); const allDone = updatedMatchups.filter((m) => !m.isBye).every((m) => m.status === "completed"); await sendJson(`${apiBase}/brackets/${bracket.id}/`, "PUT", { ...fresh, matchups: updatedMatchups, status: allDone ? "completed" : fresh.status }); await sendJson(`${apiBase}/games/${state.gameId}/`, "PUT", { score1: state.score1, score2: state.score2, quarter: state.quarter, clock: state.clock, status: "completed", team1Name: matchup.team1, team2Name: matchup.team2, }); await sendJson(`${apiBase}/games/${state.gameId}/sync_stats/`, "POST", { playerStats: [...state.homeRoster, ...state.awayRoster].map((x) => ({ player: x.player, playerName: x.playerName, teamName: x.teamName, fgm2: x.fgm2, fga2: x.fga2, fgm3: x.fgm3, fga3: x.fga3, ftm: x.ftm, fta: x.fta, oreb: x.oreb, dreb: x.dreb, assists: x.assists, steals: x.steals, blocks: x.blocks, fouls: x.fouls, turnovers: x.turnovers, points: x.points, })), }); await loadPage(); }); }; const bracketFilter = document.getElementById("games-bracket-filter"); const matchupFilter = document.getElementById("games-matchup-filter"); const syncMatchupFilter = () => { const rows = getMatchupsForBracket(selectedBracketId); if (!rows.length) { matchupFilter.innerHTML = ''; selectedMatchupId = ""; return rows; } matchupFilter.innerHTML = rows.map((m) => ``).join(""); const hasSelected = rows.some((m) => String(m.id) === String(selectedMatchupId)); selectedMatchupId = hasSelected ?
