function escapeHtml(value) { return String(value ?? '') .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"') .replace(/'/g, '''); } function teamInitials(name) { const parts = String(name || '').trim().split(/\s+/).filter(Boolean); if (parts.length === 0) return '?'; if (parts.length === 1) return parts[0].slice(0, 2).toUpperCase(); return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase(); } function predictionEmptyState(message) { return `

No predictions available

${escapeHtml(message)}

`; } function topScorerLabel(features) { const top = (features?.top_players && features.top_players[0]) || null; if (!top || !top.name) return '—'; const ppg = (top.ppg ?? 0).toFixed ? top.ppg.toFixed(1) : top.ppg; return `${escapeHtml(top.name)} (${ppg} PPG)`; } function statValue(value, suffix = '') { if (value === null || value === undefined || value === '') return '—'; return `${escapeHtml(value)}${suffix}`; } function renderPredictionCard(matchup, prediction, bracketId, bracketName = '') { const team1 = matchup.team1 || 'TBD'; const team2 = matchup.team2 || 'TBD'; const hasPrediction = !!prediction; const winner = prediction?.predictedWinner || ''; const confidence = Math.max(0, Math.min(100, Number(prediction?.confidence) || 0)); const team1IsWinner = hasPrediction && winner === team1; const team2IsWinner = hasPrediction && winner === team2; const team1Pct = !hasPrediction ? 50 : (team1IsWinner ? confidence : 100 - confidence); const team2Pct = !hasPrediction ? 50 : (team2IsWinner ? confidence : 100 - confidence); const factors = prediction?.factors || {}; const f1 = factors.team1 || {}; const f2 = factors.team2 || {}; const dateLine = matchup.scheduledDate ? `${escapeHtml(matchup.scheduledDate)}${matchup.scheduledTime ? ' · ' + escapeHtml(matchup.scheduledTime) : ''}` : 'Date TBD'; const winnerBanner = hasPrediction ? `
AI Predicted Winner
${escapeHtml(winner)}
Confidence
${confidence}%
` : `
No AI prediction yet. Click below to generate one.
`; const teamRow = (name, pct, isWinner, color) => `
${escapeHtml(teamInitials(name))}
${escapeHtml(name)} ${isWinner ? '' : ''}
${pct}%
`; const statBlock = (features, isWinner) => `
${escapeHtml(features?.name || '')}
Win Streak
${statValue(features?.win_streak ?? 0)}
Avg Height
${features?.avg_height_cm ? statValue(features.avg_height_cm, ' cm') : '—'}
Top Scorer
${topScorerLabel(features)}
`; const reasoningBlock = hasPrediction && prediction?.reasoning ? `
AI Reasoning

${escapeHtml(prediction.reasoning)}

` : ''; const buttonLabel = hasPrediction ? ' Regenerate' : ' Generate AI Prediction'; return `
${bracketName ? `${escapeHtml(bracketName)}` : ''} Round ${escapeHtml(matchup.round)} Match #${escapeHtml(matchup.matchNumber)}
${dateLine}
${winnerBanner}
${teamRow(team1, team1Pct, team1IsWinner || (!hasPrediction), 'orange')}
VS
${teamRow(team2, team2Pct, team2IsWinner, 'blue')}
${hasPrediction ? `
${statBlock(f1, team1IsWinner)} ${statBlock(f2, team2IsWinner)}
` : ''} ${reasoningBlock}
`; } async function renderPredictionsPage() { const [brackets, predictions] = await Promise.all([ fetchJson(`${apiBase}/brackets/`), fetchJson(`${apiBase}/predictions/`) ]); const headerHTML = `
${window.HTUI?.renderPageHeader({ title: 'AI Game Predictions', subtitle: 'Win streak, player performance, and roster height analyzed for each upcoming matchup.', badgeHtml: `
${brackets.length} bracket${brackets.length === 1 ? '' : 's'} available
`, }) || ''}
`; if (!brackets.length) { contentEl.innerHTML = headerHTML + predictionEmptyState('No brackets available yet. Create one from the Brackets page to start generating predictions.'); return; } const bracketOptions = brackets .map((b) => ({ id: String(b.id), name: b.name || `Bracket ${b.id}`, status: b.status || 'active', matchups: b.matchups || [], currentRound: b.currentRound || 1, totalRounds: b.totalRounds || 1, })); const predictionByKey = new Map( predictions.map((p) => [`${p.bracketId}|${p.matchupId}`, p]) ); const preferredBracket = localStorage.getItem('predictionsBracketId'); const defaultBracket = bracketOptions.find((b) => b.id === preferredBracket) || bracketOptions.find((b) => b.status === 'active') || bracketOptions[0]; contentEl.innerHTML = headerHTML + `
${window.HTUI?.renderToolbar(`
Bracket
Team Search
`) || ''}
`; const bracketSelect = document.getElementById('prediction-bracket-filter'); const teamSearch = document.getElementById('prediction-team-search'); const cardsEl = document.getElementById('prediction-cards'); const metaEl = document.getElementById('prediction-filter-meta'); const collectMatchups = () => { const selectedBracketId = bracketSelect?.value || defaultBracket.id; if (selectedBracketId !== 'all') { localStorage.setItem('predictionsBracketId', selectedBracketId); } const sourceBrackets = selectedBracketId === 'all' ? bracketOptions : bracketOptions.filter((b) => b.id === selectedBracketId); return sourceBrackets.flatMap((b) => (b.matchups || []).map((m) => ({ ...m, __bracket: b }))); }; const applyFilters = () => { const q = String(teamSearch?.value || '').toLowerCase().trim(); const rows = collectMatchups().filter((m) => m.team1 && m.team2 && m.status === 'pending'); const filtered = rows.filter((m) => { if (q && !(`${m.team1} ${m.team2}`.toLowerCase().includes(q))) return false; return true; }); metaEl.textContent = `${filtered.length} matchups shown`; if (!filtered.length) { cardsEl.innerHTML = predictionEmptyState('No matchups match your current filter.'); return; } cardsEl.innerHTML = filtered.map((m) => { const key = `${m.__bracket.id}|${m.id}`; const prediction = predictionByKey.get(key); return renderPredictionCard(m, prediction, m.__bracket.id, m.__bracket.name); }).join(''); cardsEl.querySelectorAll('[data-generate]').forEach((btn) => { btn.addEventListener('click', async () => { const matchupId = btn.getAttribute('data-generate'); const bracketId = btn.getAttribute('data-bracket-id'); const originalHTML = btn.innerHTML; btn.disabled = true; btn.classList.add('opacity-70', 'cursor-not-allowed'); btn.innerHTML = ` Analyzing matchup with llama3.2... `; try { await sendJson(`${apiBase}/predictions/generate/`, 'POST', { matchupId, bracketId }); setStatus('AI prediction generated.'); await renderPredictionsPage(); } catch (err) { btn.innerHTML = originalHTML; btn.disabled = false; btn.classList.remove('opacity-70', 'cursor-not-allowed'); setStatus(`AI prediction failed: ${err.message}. Set GROQ_API_KEY on the server and check Groq model names.`, true); } }); }); }; bracketSelect?.addEventListener('change', applyFilters); teamSearch?.addEventListener('input', applyFilters); applyFilters(); }
